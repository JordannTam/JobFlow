### types/index.ts â€” Single source of truth
```typescript
/**
 * Represents a job application entry.
 * 
 * Fields:
 *     id (string)           - UUID, generated by backend
 *     company (string)      - Company name
 *     role (string)         - Job title/position
 *     status (string)       - One of: 'applied', 'interview', 'offer', 'rejected'
 *     link (string)         - Optional, URL to job posting
 *     dateApplied (string)  - Optional, ISO date string
 *     notes (string)        - Optional, personal notes
 *     description (string)  - Optional, job description for AI resume generation
 */
export interface JobApplication {
  id: string;
  company: string;
  role: string;
  status: 'applied' | 'interview' | 'offer' | 'rejected';
  link?: string;
  dateApplied?: string;
  notes?: string;
  description?: string;
}

export type ApplicationStatus = 'applied' | 'interview' | 'offer' | 'rejected';

export type CreateApplicationInput = Omit<JobApplication, 'id'>;
export type UpdateApplicationInput = Omit<JobApplication, 'id'>;
```

### services/api.ts â€” All API calls in one place
```typescript
import { API_URL } from '../constants';
import { JobApplication, CreateApplicationInput, UpdateApplicationInput } from '../types';

/**
 * API service for job application CRUD operations.
 * 
 * All functions return Promises and handle JSON parsing.
 * Errors are thrown on non-2xx responses.
 */
export const api = {
  /**
   * Fetches all job applications.
   * 
   * Arguments: None
   * 
   * Return Value:
   *     Returns Promise<JobApplication[]> on success
   *     Throws Error on network/server failure
   */
  getApplications: async (): Promise<JobApplication[]> => {
    const res = await fetch(`${API_URL}/applications`);
    const data = await res.json();
    return JSON.parse(data.body);
  },

  /**
   * Creates a new job application.
   * 
   * Arguments:
   *     data (CreateApplicationInput) - Application data without id
   * 
   * Return Value:
   *     Returns Promise<{ message: string }> on success
   *     Throws Error on validation/server failure
   */
  createApplication: async (data: CreateApplicationInput): Promise<{ message: string }> => {
    const res = await fetch(`${API_URL}/applications`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
    return res.json();
  },

  /**
   * Updates an existing job application.
   * 
   * Arguments:
   *     id (string)                    - Application UUID
   *     data (UpdateApplicationInput)  - Updated application data
   * 
   * Return Value:
   *     Returns Promise<{ message: string }> on success
   *     Throws Error on validation/server failure
   */
  updateApplication: async (id: string, data: UpdateApplicationInput): Promise<{ message: string }> => {
    const res = await fetch(`${API_URL}/applications/${id}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
    return res.json();
  },

  /**
   * Deletes a job application.
   * 
   * Arguments:
   *     id (string) - Application UUID to delete
   * 
   * Return Value:
   *     Returns Promise<{ message: string }> on success
   *     Throws Error on server failure
   */
  deleteApplication: async (id: string): Promise<{ message: string }> => {
    const res = await fetch(`${API_URL}/applications/${id}`, {
      method: 'DELETE',
    });
    return res.json();
  },
};
```
